-- Teleport function
local function teleportTo(location)
    pcall(function()
        local char = lp.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.CFrame = CFrame.new(location.x, location.y, location.z)
        end
    end)
end

-- ============ RESPAWN FUNCTION ============
local function respawnPlayer()
    pcall(function()
        local char = lp.Character
        if char then
            local humanoid = char:FindFirstChild("Humanoid")
            if humanoid then
                humanoid.Health = 0
            end
        end
    end)
end

-- ============ SPEED BUTTONS (RELIABLE) ============
local function setSpeed(speedValue)
    speedValue = math.clamp(speedValue, 1, 100)
    getgenv().HAPPYY_Speed = speedValue
    
    -- Multiple methods to apply speed
    pcall(function()
        local char = lp.Character
        if char and char:FindFirstChild("Humanoid") then
            char.Humanoid.WalkSpeed = speedValue
        end
    end)
    
    -- Look for speed values in character
    pcall(function()
        local char = lp.Character
        if char then
            for _, child in ipairs(char:GetChildren()) do
                if child:IsA("NumberValue") and child.Name:lower():find("speed") then
                    child.Value = speedValue
                end
            end
        end
    end)
    
    print("Speed set to: " .. speedValue)
end

-- Aggressive speed keep (every frame)

    if getgenv().HAPPYY_Speed ~= 25 then
        pcall(function()
            local char = lp.Character
            if char and char:FindFirstChild("Humanoid") then
                if char.Humanoid.WalkSpeed ~= getgenv().HAPPYY_Speed then
                    char.Humanoid.WalkSpeed = getgenv().HAPPYY_Speed
                end
            end
        end)
    end
end)

-- ============ PLAYER ESP ============
local espObjects = {}
local espEnabled = false

local function getDistance(player)
    local char = lp.Character
    local targetChar = player.Character
    if char and targetChar then
        local hrp = char:FindFirstChild("HumanoidRootPart")
        local targetHrp = targetChar:FindFirstChild("HumanoidRootPart")
        if hrp and targetHrp then
            local distance = (hrp.Position - targetHrp.Position).Magnitude
            return math.floor(distance)
        end
    end
    return nil
end

local function createPlayerESP(player)
    if player == lp then return nil end
    
    local targetChar = player.Character
    if not targetChar or not targetChar.Parent then return nil end
    
    local highlight = Instance.new("Highlight")
    highlight.Name = "PlayerESP_" .. player.Name
    highlight.Adornee = targetChar
    highlight.FillColor = Color3.fromRGB(255, 255, 255)
    highlight.FillTransparency = 0.5
    highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
    highlight.OutlineTransparency = 0.2
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Parent = targetChar
    
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "PlayerLabel_" .. player.Name
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 180, 0, 55)
    billboard.StudsOffset = Vector3.new(0, 2.5, 0)
    billboard.Adornee = targetChar
    billboard.Parent = targetChar
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 1, 0)
    frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    frame.BackgroundTransparency = 0.6
    frame.BorderSizePixel = 1
    frame.BorderColor3 = Color3.fromRGB(255, 255, 255)
    frame.Parent = billboard
    
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 0.55, 0)
    nameLabel.Position = UDim2.new(0, 0, 0, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = player.Name
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextScaled = true
    nameLabel.Font = Enum.Font.GothamBold
    nameLabel.Parent = frame
    
    local distanceLabel = Instance.new("TextLabel")
    distanceLabel.Size = UDim2.new(1, 0, 0.45, 0)
    distanceLabel.Position = UDim2.new(0, 0, 0.55, 0)
    distanceLabel.BackgroundTransparency = 1
    distanceLabel.TextColor3 = Color3.fromRGB(255, 255, 100)
    distanceLabel.TextScaled = true
    distanceLabel.Font = Enum.Font.Gotham
    distanceLabel.Parent = frame
    
    local connection
    connection = RunService.RenderStepped:Connect(function()
        if not espEnabled or not billboard or not billboard.Parent then
            if connection then connection:Disconnect() end
            return
        end
        
        local dist = getDistance(player)
        if dist then
            distanceLabel.Text = dist .. " meters"
        else
            distanceLabel.Text = "out of range"
        end
    end)
    
    return {highlight = highlight, billboard = billboard, connection = connection, player = player}
end

local function clearPlayerESP()
    for _, obj in pairs(espObjects) do
        pcall(function()
            if obj.highlight then obj.highlight:Destroy() end
            if obj.billboard then obj.billboard:Destroy() end
            if obj.connection then obj.connection:Disconnect() end
        end)
    end
    espObjects = {}
end

local function refreshPlayerESP()
    if not espEnabled then return end
    
    clearPlayerESP()
    
    for _, player in ipairs(game:GetService("Players"):GetPlayers()) do
        if player ~= lp and player.Character and player.Character.Parent then
            local esp = createPlayerESP(player)
            if esp then
                table.insert(espObjects, esp)
            end
        end
    end
    
    print("Player ESP refreshed - Tracking " .. #espObjects .. " players")
end

local function togglePlayerESP(state)
    espEnabled = state
    if espEnabled then
        refreshPlayerESP()
    else
        clearPlayerESP()
        print("Player ESP disabled")
    end
end

game:GetService("Players").PlayerAdded:Connect(function(player)
    if espEnabled then
        task.wait(0.5)
        if player.Character and player.Character.Parent then
            local esp = createPlayerESP(player)
            if esp then
                table.insert(espObjects, esp)
            end
        end
    end
end)

game:GetService("Players").PlayerRemoving:Connect(function(player)
    if espEnabled then
        for i, obj in pairs(espObjects) do
            if obj.player == player then
                pcall(function()
                    if obj.highlight then obj.highlight:Destroy() end
                    if obj.billboard then obj.billboard:Destroy() end
                    if obj.connection then obj.connection:Disconnect() end
                end)
                table.remove(espObjects, i)
                break
            end
        end
    end
end)

-- ============ AUTO CAST ============
local isFishing = false
local isCasting = false
local castCooldown = false

RunService.RenderStepped:Connect(function()
    pcall(function()
        local char = lp.Character
        if char and char:GetAttribute("Fishing") then
            isFishing = true
        else
            isFishing = false
        end
    end)
end)

task.spawn(function()
    while task.wait(1.5) do
        if getgenv().HAPPYY_AutoCast and not isCasting and not castCooldown and not isFishing then
            isCasting = true
            pcall(function()
                local Events = RS:FindFirstChild("Events") or RS
                local FishingRemote = Events:FindFirstChild("Fishing") or Events:FindFirstChild("Cast")
                if FishingRemote then
                    FishingRemote:FireServer()
                    castCooldown = true
                    task.wait(2)
                    castCooldown = false
                end
            end)
            task.wait(1)
            isCasting = false
        end
        task.wait()
    end
end)

-- ============ AUTO SKILLS ============
task.spawn(function()
    local lastUse = {Z=0, X=0, V=0, C=0}
    while task.wait(0.3) do
        local now = tick()
        
        if getgenv().HAPPYY_SkillZ and (now - lastUse.Z) >= 1.5 then
            pcall(function()
                local VirtualInput = game:GetService("VirtualInputManager")
                VirtualInput:SendKeyEvent(true, Enum.KeyCode.Z, false, game)
                task.wait(0.05)
                VirtualInput:SendKeyEvent(false, Enum.KeyCode.Z, false, game)
                lastUse.Z = now
            end)
        end
        
        if getgenv().HAPPYY_SkillX and (now - lastUse.X) >= 1.5 then
            pcall(function()
                local VirtualInput = game:GetService("VirtualInputManager")
                VirtualInput:SendKeyEvent(true, Enum.KeyCode.X, false, game)
                task.wait(0.05)
                VirtualInput:SendKeyEvent(false, Enum.KeyCode.X, false, game)
                lastUse.X = now
            end)
        end
        
        if getgenv().HAPPYY_SkillV and (now - lastUse.V) >= 1.5 then
            pcall(function()
                local VirtualInput = game:GetService("VirtualInputManager")
                VirtualInput:SendKeyEvent(true, Enum.KeyCode.V, false, game)
                task.wait(0.05)
                VirtualInput:SendKeyEvent(false, Enum.KeyCode.V, false, game)

-- ============ ANCHOR / AUTO REEL ============

        pcall(function()
            local fishingUI = lp.PlayerGui:FindFirstChild("MainGui")
            if fishingUI then

                if fishing and fishing.Visible then
                    local barFrame = fishing:FindFirstChild("BarFrame")
                    if barFrame then
                        local bar = barFrame:FindFirstChild("Bar")
                        if bar then
                            bar.Position = UDim2.new(0.5, 0, bar.Position.Y.Scale, 0)
                        end
                    end
                end
            end
        end)
    end
end)

end
