-- HAPPYY SCRIPT v11.0 - BOSS LOCK (Stops checking after boss detected)
local DrRayLibrary = loadstring(game:HttpGet("https://raw.githubusercontent.com/AZYsGithub/DrRay-UI-Library/main/DrRay.lua"))()
local window = DrRayLibrary:Load("HAPPYY SCRIPT", "Default")

local lp = game:GetService("Players").LocalPlayer
local rs = game:GetService("ReplicatedStorage")
local runService = game:GetService("RunService")
local players = game:GetService("Players")
local vim = game:GetService("VirtualInputManager")

-- Settings
local settings = {
    anchor = false,
    autoCast = false,
    autoBoss = false,
    speed = 25,
    esp = false,
    skillZ = false,
    skillX = false,
    skillV = false,
    skillC = false
}

-- State
local resetting = false
local casting = false
local castCD = false
local bossDetected = false
local bossLocked = false  -- NEW: Prevents further checks once boss is confirmed

-- Find remotes
local castRemote = nil
local sellRemote = nil

for _, name in ipairs({"Fishing", "Cast", "StartFishing"}) do
    castRemote = rs:FindFirstChild(name) or (rs:FindFirstChild("Events") and rs.Events:FindFirstChild(name))
    if castRemote then break end
end

for _, name in ipairs({"SellAll", "Sell", "SellFish"}) do
    sellRemote = rs:FindFirstChild(name) or (rs:FindFirstChild("Events") and rs.Events:FindFirstChild(name))
    if sellRemote then break end
end

    {"Beginning Isle", -240.01, 5.56, 40.94},

    {"Perch Isle", 31.89, 6.20, 1227.51},

local function teleportTo(x, y, z)
    local char = lp.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        char.HumanoidRootPart.CFrame = CFrame.new(x, y, z)
    end
end

-- Speed
local function setSpeed(v)
    v = math.clamp(v, 1, 100)
    settings.speed = v
    local char = lp.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid.WalkSpeed = v
    end
    print("Speed: " .. v)
end

lp.CharacterAdded:Connect(function(char)
    task.wait(0.5)
    local hum = char:FindFirstChild("Humanoid")
    if hum then hum.WalkSpeed = settings.speed end
end)

runService.RenderStepped:Connect(function()
    if settings.speed ~= 25 then
        local char = lp.Character
        if char and char:FindFirstChild("Humanoid") then
            local hum = char.Humanoid
            if hum.WalkSpeed ~= settings.speed then
                hum.WalkSpeed = settings.speed
            end
        end
    end
end)

-- OLD ANCHOR
runService.RenderStepped:Connect(function()
    if settings.anchor then
        pcall(function()
            local fishingUI = lp.PlayerGui:FindFirstChild("MainGui")
            if fishingUI then
                local fishing = fishingUI:FindFirstChild("Fishing")
                if fishing and fishing.Visible then
                    local barFrame = fishing:FindFirstChild("BarFrame")
                    if barFrame then
                        local bar = barFrame:FindFirstChild("Bar")
                        if bar then
                            bar.Position = UDim2.new(0.5, 0, bar.Position.Y.Scale, 0)
                        end
                    end
                end
            end
        end)
    end
end)

-- Reset rod (press 1 twice)
local function resetRod()
    pcall(function()
        print("[Auto Boss] Resetting rod...")
        vim:SendKeyEvent(true, Enum.KeyCode.One, false, game)
        task.wait(0.1)
        vim:SendKeyEvent(false, Enum.KeyCode.One, false, game)
        task.wait(0.2)
        vim:SendKeyEvent(true, Enum.KeyCode.One, false, game)
        task.wait(0.1)
        vim:SendKeyEvent(false, Enum.KeyCode.One, false, game)
    end)
end

-- GET COUNTDOWN NUMBER (only gets the FIRST number, ignores after boss detected)
local function getCountdownNumber()
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("TextLabel") then
            local text = obj.Text or ""
            -- Only match if it's specifically "boss fight" or "minigame" with a number
            local bossNum = string.match(text, "boss fight will start in (%d+)")
            if bossNum then
                return tonumber(bossNum)
            end
            local minigameNum = string.match(text, "minigame will start in (%d+)")
            if minigameNum then
                return tonumber(minigameNum)
            end
        end
    end
    
    local gui = lp.PlayerGui
    if gui then
        for _, obj in ipairs(gui:GetDescendants()) do
            if obj:IsA("TextLabel") then
                local text = obj.Text or ""
                local bossNum = string.match(text, "boss fight will start in (%d+)")
                if bossNum then
                    return tonumber(bossNum)
                end
                local minigameNum = string.match(text, "minigame will start in (%d+)")
                if minigameNum then
                    return tonumber(minigameNum)
                end
            end
        end
    end
    return nil
end

-- Get fishing UI
local function getFishingUI()
    local gui = lp.PlayerGui:FindFirstChild("MainGui")
    if gui then
        return gui:FindFirstChild("Fishing")
    end
    return nil
end

-- AUTO BOSS LOOP (with boss locking)
task.spawn(function()
    while true do
        task.wait(0.3)
        
        if not settings.autoBoss then
            resetting = false
            bossDetected = false
            bossLocked = false
        end
        
        local fishing = getFishingUI()
        local uiVisible = fishing and fishing.Visible
        
        if uiVisible and settings.autoBoss and not resetting then
            
            -- ONLY check if boss is NOT already locked
            if not bossLocked then
                local countdownNum = getCountdownNumber()
                
                if countdownNum then
                    -- BOSS DETECTED (10, 9, or 8)
                    if countdownNum == 10 or countdownNum == 9 or countdownNum == 8 then
                        bossDetected = true
                        bossLocked = true  -- LOCK: No more checks for this fight
                        print("[BOSS] 🎯 COUNTDOWN " .. countdownNum .. "s DETECTED!")
                        print("[BOSS] LOCKED IN - No auto reset will happen")
                        print("[BOSS] Make sure ANCHOR is ON to catch the boss!")
                    end
                    
                    -- REGULAR FISH (3 or 2) - reset immediately
                    if countdownNum == 3 or countdownNum == 2 then
                        print("[Regular] COUNTDOWN " .. countdownNum .. "s DETECTED - Resetting rod...")
                        resetting = true
                        resetRod()
                        task.wait(0.5)
                        resetting = false
                    end
                end
            end
            
        elseif not uiVisible then
            -- UI closed - reset all states for next cast
            bossDetected = false
            bossLocked = false
            resetting = false
        end
    end
end)

-- Auto Cast
task.spawn(function()
    while true do
        task.wait(0.5)
        
        if settings.autoBoss and castRemote then
            local fishing = getFishingUI()
            local char = lp.Character
            local isFishing = char and char:GetAttribute("Fishing") == true
            
            if not isFishing and not casting and not castCD and not resetting and not (fishing and fishing.Visible) then
                casting = true
                pcall(function() castRemote:FireServer() end)
                print("[Auto Boss] Casting rod...")
                castCD = true
                task.wait(2)
                castCD = false
                casting = false
            end
        elseif settings.autoCast and castRemote then
            local char = lp.Character
            local isFishing = char and char:GetAttribute("Fishing") == true
            
            if not isFishing and not casting and not castCD then
                casting = true
                pcall(function() castRemote:FireServer() end)
                castCD = true
                task.wait(2)
                castCD = false
                casting = false
            end
        end
    end
end)

-- Auto Skills
task.spawn(function()
    local last = {Z = 0, X = 0, V = 0, C = 0}
    while true do
        task.wait(0.3)
        local now = tick()
        
        if settings.skillZ and now - last.Z >= 1.5 then
            vim:SendKeyEvent(true, Enum.KeyCode.Z, false, game)
            task.wait(0.05)
            vim:SendKeyEvent(false, Enum.KeyCode.Z, false, game)
            last.Z = now
        end
        
        if settings.skillX and now - last.X >= 1.5 then
            vim:SendKeyEvent(true, Enum.KeyCode.X, false, game)
            task.wait(0.05)
            vim:SendKeyEvent(false, Enum.KeyCode.X, false, game)
            last.X = now
        end
        
        if settings.skillV and now - last.V >= 1.5 then
            vim:SendKeyEvent(true, Enum.KeyCode.V, false, game)
            task.wait(0.05)
            vim:SendKeyEvent(false, Enum.KeyCode.V, false, game)
            last.V = now
        end
        
        if settings.skillC and now - last.C >= 1.5 then
            vim:SendKeyEvent(true, Enum.KeyCode.C, false, game)
            task.wait(0.05)
            vim:SendKeyEvent(false, Enum.KeyCode.C, false, game)
            last.C = now
        end
    end
end)

-- ESP
local espEnabled = false
local espHighlights = {}

local function toggleESP(state)
    espEnabled = state
    if not state then
        for _, h in pairs(espHighlights) do
            pcall(function() h:Destroy() end)
        end
        espHighlights = {}
        return
    end
    
    for _, player in ipairs(players:GetPlayers()) do
        if player ~= lp and player.Character then
            local h = Instance.new("Highlight")
            h.Adornee = player.Character
            h.FillColor = Color3.new(1, 1, 1)
            h.FillTransparency = 0.5
            h.OutlineColor = Color3.new(1, 0, 0)
            h.Parent = player.Character
            table.insert(espHighlights, h)
        end
    end
end

players.PlayerAdded:Connect(function(p)
    if espEnabled and p ~= lp then
        p.CharacterAdded:Connect(function(char)
            task.wait(0.5)
            local h = Instance.new("Highlight")
            h.Adornee = char
            h.FillColor = Color3.new(1, 1, 1)
            h.FillTransparency = 0.5
            h.OutlineColor = Color3.new(1, 0, 0)
            h.Parent = char
            table.insert(espHighlights, h)
        end)
    end
end)

-- Sell
local function sellAll()
    if sellRemote then
        pcall(function() sellRemote:FireServer() end)
        print("Sold all!")
    else
        print("Sell remote not found")
    end
end

local function respawn()
    local char = lp.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid.Health = 0
    end
end
